% Columns

% Two
\newcommand{\myTwoColumns}[4]{
\begin{columns}[t]
\begin{column}[t]{#1\linewidth}
#3
\end{column}
\begin{column}[t]{#2\linewidth}
#4
\end{column}
\end{columns}
}

% Three
\newcommand{\myThreeColumns}[6]{
\begin{columns}[t]
\begin{column}[t]{#1\linewidth}
#4
\end{column}
\begin{column}[t]{#2\linewidth}
#5
\end{column}
\begin{column}[t]{#3\linewidth}
#6
\end{column}
\end{columns}
}

% Box with title and text

% usual
\newcommand{\myBlock}[2]{
\bigskip
\begin{block}{\large #1}
#2
\end{block}
}

% alert
\newcommand{\myAlertBlock}[2]{
\bigskip
\begin{alertblock}{\large #1}
#2
\end{alertblock}
}

% Text block
\newcommand{\myTextBlock}[1]{
\bigskip

#1
}

% Insert picture to text. 
\newcommand{\myPic}[3]{
\begin{figure}[H]
\centering\par\bigskip \includegraphics[scale=#2]{#3}
\caption{#1}
\label{#3}
\end{figure}
}

#1
\end{itemize}
}

% Enumerated

\begin{enumerate}
#1

}
